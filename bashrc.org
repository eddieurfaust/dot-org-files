*** cl()
    #+BEGIN_SRC sh :tangle ~/.bashrc
      # cd and ls in one
      cl() {
          dir=$1
          if [[ -z "$dir" ]]; then
              dir=$HOME
          fi
          if [[ -d "$dir" ]]; then
              cd "$dir"
              ls
          else
              echo "bash: cl: '$dir': Directory not found"
          fi
      }
    #+END_SRC
